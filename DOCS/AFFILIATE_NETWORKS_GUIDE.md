### **Step 2: Integrate Affiliate Links**

**Backend Implementation:**

```typescript
// backend/src/utils/affiliate-links.ts

interface AffiliateConfig {
  network: 'amazon' | 'aliexpress' | 'ebay' | 'awin' | 'other';
  country: string;
  affiliateId: string;
  tag: string;
}

const AFFILIATE_CONFIGS: AffiliateConfig[] = [
  {
    network: 'amazon',
    country: 'US',
    affiliateId: 'your-amazon-us-id',
    tag: 'yourapp-20',
  },
  {
    network: 'amazon',
    country: 'UK',
    affiliateId: 'your-amazon-uk-id',
    tag: 'yourapp-21',
  },
  {
    network: 'aliexpress',
    country: 'GLOBAL',
    affiliateId: 'your-aliexpress-id',
    tag: 'yourapp',
  },
  // Add more configs...
];

export function addAffiliateTag(url: string, userCountry: string): string {
  const domain = new URL(url).hostname;
  
  // Amazon
  if (domain.includes('amazon')) {
    const config = AFFILIATE_CONFIGS.find(
      c => c.network === 'amazon' && domain.includes(c.country.toLowerCase())
    );
    if (config) {
      return `${url}${url.includes('?') ? '&' : '?'}tag=${config.tag}`;
    }
  }
  
  // AliExpress
  if (domain.includes('aliexpress')) {
    const config = AFFILIATE_CONFIGS.find(c => c.network === 'aliexpress');
    if (config) {
      return `${url}${url.includes('?') ? '&' : '?'}aff_platform=portals-tool&aff_trace_key=${config.affiliateId}`;
    }
  }
  
  // eBay
  if (domain.includes('ebay')) {
    const config = AFFILIATE_CONFIGS.find(c => c.network === 'ebay');
    if (config) {
      return `${url}${url.includes('?') ? '&' : '?'}mkcid=1&mkrid=${config.affiliateId}`;
    }
  }
  
  // Return original URL if no affiliate config found
  return url;
}
```

**Frontend Integration:**

```typescript
// utils/openStoreLink.ts

import { addAffiliateTag } from '@/utils/api';

export async function openStoreLink(url: string, options?: {
  source?: string;
  storeDomain?: string;
}) {
  try {
    // Add affiliate tag via backend
    const { authenticatedPost } = await import('@/utils/api');
    const response = await authenticatedPost<{ affiliateUrl: string }>('/api/affiliate/add-tag', {
      url,
      source: options?.source || 'app',
    });
    
    const affiliateUrl = response.affiliateUrl || url;
    
    // Open in browser
    await Linking.openURL(affiliateUrl);
    
    // Track click
    console.log('[Affiliate] Opened link:', affiliateUrl);
  } catch (error) {
    console.error('[Affiliate] Error opening link:', error);
    // Fallback to original URL
    await Linking.openURL(url);
  }
}
```

### **Step 3: Track Conversions**

**Backend Endpoint:**

```typescript
// backend/src/routes/affiliate.ts

app.post('/api/affiliate/track-click', async (request, reply) => {
  const { itemId, url, network, country } = request.body;
  const userId = request.user.id;
  
  // Save click to database
  await db.insert(schema.affiliateClicks).values({
    userId,
    itemId,
    url,
    network,
    country,
    clickedAt: new Date().toISOString(),
  });
  
  return { success: true };
});

app.post('/api/affiliate/track-conversion', async (request, reply) => {
  const { orderId, amount, currency, network } = request.body;
  const userId = request.user.id;
  
  // Save conversion to database
  await db.insert(schema.affiliateConversions).values({
    userId,
    orderId,
    amount,
    currency,
    network,
    convertedAt: new Date().toISOString(),
  });
  
  return { success: true };
});
```

## ðŸ“Š Best Practices

### **1. Disclose Affiliate Relationships**
- Add disclaimer in app settings
- Comply with FTC guidelines
- Be transparent with users

### **2. Optimize for User Experience**
- Don't spam affiliate links
- Only link to quality products
- Provide value first, monetize second

### **3. Test and Iterate**
- A/B test different networks
- Monitor conversion rates
- Optimize for highest revenue

### **4. Diversify Networks**
- Don't rely on one network
- Have backups for each country
- Adapt to local preferences

### **5. Stay Compliant**
- Follow network terms of service
- Don't use incentivized clicks
- Avoid cookie stuffing
